[MeasureItem$0Title]

Plugin=WebParser
Url=[MeasureSite]
RegExp="(?siU)#Title#"
StringIndex=<% $rowNumOne + 1 %>
StringIndex2=1
Substitute=#Sub#

[MeasureItem$0Link]
Measure=Plugin
Plugin=WebParser
Url=[MeasureSite]
RegExp="(?siU)#Link#"
StringIndex=<% $rowNumOne + 1 %>
StringIndex2=1

[MeasureItem$0Percent]
Measure=Plugin
Plugin=WebParser
Url=[MeasureSite]
RegExp="(?siU)#Percent#"
StringIndex=<% $rowNumOne + 1 %>
StringIndex2=1
IfEqualValue=0
IfEqualAction=[!HideMeter MeterItem$0Right][!SetOption MeterItem$0Title ToolTipHidden 1][!Redraw]
IfAboveValue=0
IfAboveAction=[!ShowMeter MeterItem$0Right][!SetOption MeterItem$0Title ToolTipHidden 0][!Redraw]

[Prc$0]
Measure=Calc
Formula=MeasureItem$0Percent/100
Group=Calc
Disabled=1

[MeasureItem$0Price]
Measure=Plugin
Plugin=WebParser
Url=[MeasureSite]
RegExp="(?siU)#Price#"
StringIndex=<% $rowNumOne + 1 %>
StringIndex2=1
RegExpSubstitute=1
Substitute=#Sub#
